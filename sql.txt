SELECT 
    ID_ALUNO,
    NOME_ALUNO,
    CURSO.DESCR AS CURSO,
    SITUACAO_CONTRATO,
    CASE 
        WHEN TURNO = 'NOI' THEN 'NOITE' 
        WHEN TURNO = 'MAN' THEN 'MANHÃƒ' 
        WHEN TURNO = 'MAD' THEN 'MADRUGADA' 
        WHEN TURNO = 'TAR' THEN 'TARDE' 
    END AS TURNO2,
    TURMA_BASE,
    'CLIQUE_PARA_COLOCAR_O_CODIGO_DA_PROVA' AS COD_PROVA 
FROM 
    V_TICKET_CONTRATO 
INNER JOIN 
    CURSO ON CURSO.CURSO = V_TICKET_CONTRATO.ID_CURSO 
WHERE 
    (CLIQUE_PARA_COLOCAR_AS_TURMAS)
    AND SITUACAO_CONTRATO = 'Confirmado' 
    AND TIPO_CONTRATO = 'Normal' 
ORDER BY 
    TURMA_BASE;
